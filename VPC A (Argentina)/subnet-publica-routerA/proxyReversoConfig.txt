upstream backends {
    #backend1
    server 10.1.10.26:88;
    #backend2
    server 10.1.10.199:88;
}

server {
    listen 80;
    listen [::]:80;
    access_log /var/log/nginx/reverse-access.log;  
    error_log /var/log/nginx/reverse-error.log;
    
    # 1. Web local del Router A
    location / {
        root /var/www/html;
        index index.html;
        try_files $uri $uri/ =404;
    }

    # 2. Proxy hacia los Backends
    location /intranet/ {
        proxy_pass http://backends/;
        # Cabeceras para pasar la IP real del cliente al backend
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
